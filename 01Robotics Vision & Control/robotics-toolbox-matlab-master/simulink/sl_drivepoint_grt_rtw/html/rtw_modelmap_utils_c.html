   35   /** Function: rtwCAPI_EncodePath ===============================================
   36    *  Abatract:
   37    *     Escape all '|' characters in bpath. For examples 'aaa|b' will become
   38    *     'aaa~|b'. The caller is responsible for freeing the returned string
   39    *
   40    *
   41    * NOTE: returned string can be NULL in two cases:
   42    *     (1) string passed in was NULL
   43    *     (2) a memory allocation error occurred
   44    * In the second case, the caller need to report the error
   45    */
   46   char* rtwCAPI_EncodePath(const char* path)
   47   {
   48       char* encodedPath     = NULL;
   49       size_t pathLen        = (path==NULL) ? 0:strlen(path) + 1;
   50       size_t encodedPathLen = pathLen;
   51       unsigned i;
   52       unsigned j = 0;
   53   
   54       if (path == NULL) return NULL;
   55   
   56       for (i = 0; i < pathLen; ++i) {
   57           if (path[i] == '|' || path[i] == '~') ++encodedPathLen;
   58       }
   59   
   60       encodedPath = (char_T*)utMalloc(encodedPathLen*sizeof(char_T));
   61       if (encodedPath == NULL) return encodedPath;
   62   
   63       for (i = 0; i < pathLen; ++i) {
   64           char ch = path[i];
   65           if (ch == '~' || ch == '|') encodedPath[j++] = '~';
   66           encodedPath[j++] = ch;
   67       }
   68       utAssert(j == encodedPathLen);
   69       utAssert(encodedPath[j-1] == '\0');
   70   
   71       return encodedPath;
   72   
   73   } /* rtwCAPI_EncodePath */

   75   /** Function: rtwCAPI_GetSigAddrFromMap ========================================
   76    *
   77    */
   78   void rtwCAPI_GetSigAddrFromMap(uint8_T      isPointer,
   79                                  int_T*       sigComplexity,
   80                                  int_T*       sigDataType,
   81                                  void**       sigDataAddr,
   82                                  int_T*       sigIdx,
   83                                  uint_T       mapIdx,
   84                                  void**       dataAddrMap)
   85   {
   86       if (isPointer) {
   87           /* Dereference pointer and cache the address */
   88   
   89           /* Imported Pointers cannot be complex - Assert */
   90           utAssert(sigComplexity[*sigIdx] != 1);
   91           UNUSED_PARAMETER(sigComplexity);
   92   
   93           /* Check for data type and dereference accordingly */
   94           switch (sigDataType[*sigIdx]) {
   95             case SS_DOUBLE:
   96               sigDataAddr[*sigIdx] = \
   97                   (void*) *((real_T **) dataAddrMap[mapIdx]);
   98               break;
   99             case SS_SINGLE:
  100               sigDataAddr[*sigIdx] = \
  101                   (void*) *((real32_T **) dataAddrMap[mapIdx]);
  102               break;
  103             case SS_UINT32:
  104               sigDataAddr[*sigIdx] = \
  105                   (void*) *((uint32_T **) dataAddrMap[mapIdx]);
  106               break;
  107             case SS_INT32:
  108               sigDataAddr[*sigIdx] = \
  109                   (void*) *((int32_T **) dataAddrMap[mapIdx]);
  110               break;
  111             case SS_UINT16:
  112               sigDataAddr[*sigIdx] = \
  113                   (void*) *((uint16_T **) dataAddrMap[mapIdx]);
  114               break;
  115             case SS_INT16:
  116               sigDataAddr[*sigIdx] = \
  117                   (void*) *((int16_T **) dataAddrMap[mapIdx]);
  118               break;
  119             case SS_UINT8:
  120               sigDataAddr[*sigIdx] = \
  121                   (void*) *((uint8_T **) dataAddrMap[mapIdx]);
  122               break;
  123             case SS_INT8:
  124               sigDataAddr[*sigIdx] = \
  125                   (void*) *((int8_T **) dataAddrMap[mapIdx]);
  126               break;
  127             case SS_BOOLEAN:
  128               sigDataAddr[*sigIdx] = \
  129                   (void*) *((boolean_T **) dataAddrMap[mapIdx]);
  130               break;
  131             default:
  132               sigDataAddr[*sigIdx] = \
  133                   (void*) *((real_T **) dataAddrMap[mapIdx]);
  134               break;
  135           }  /* end switch */
  136       } else {
  137           /* if Data is not a pointer store the address directly */
  138           sigDataAddr[*sigIdx] = dataAddrMap[mapIdx];
  139       }
  140   
  141   } /* rtwCAPI_GetSigAddrFromMap */

  147   boolean_T rtwCAPI_HasStates(const rtwCAPI_ModelMappingInfo* mmi)
  148   {
  149       int_T i;
  150       int_T nCMMI;
  151   
  152       if (mmi == NULL) return(0U);
  153   
  154       if (rtwCAPI_GetNumStates(mmi) > 0) return(1U);
  155   
  156       nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);
  157       for (i = 0; i < nCMMI; ++i) {
  158           if (rtwCAPI_HasStates(rtwCAPI_GetChildMMI(mmi,i))) return(1U);
  159       }
  160       return(0U);
  161   
  162   } /* rtwCAPI_HasStates */

  224   /** Function: rtwCAPI_GetNumContStateRecords ===================================
  225    *
  226    */
  227   int_T rtwCAPI_GetNumContStateRecords(const rtwCAPI_ModelMappingInfo* mmi)
  228   {
  229       int_T i;
  230       int_T nRecs;
  231       int_T nCMMI;
  232       int_T nCStateRecs;
  233       const rtwCAPI_States* states;
  234       const rtwCAPI_DataTypeMap* dataTypeMap;
  235   
  236       if (mmi == NULL) return 0;
  237   
  238       nCStateRecs = 0;
  239       states      = rtwCAPI_GetStates(mmi);
  240       dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);
  241   
  242       nRecs = rtwCAPI_GetNumStates(mmi);
  243       for (i = 0; i < nRecs; i++) {
  244           if (rtwCAPI_IsAContinuousState(states,i)) {
  245               ++nCStateRecs;
  246   
  247               /* All continuous states should be able to be logged to MAT-File
  248                * so we do not need to skip any states here. */
  249               utAssert(rtwCAPI_CanLogStateToMATFile(dataTypeMap, states, i));
  250           }
  251       }
  252   
  253       nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);
  254       for (i = 0; i < nCMMI; ++i) {
  255           const rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);
  256           nCStateRecs += rtwCAPI_GetNumContStateRecords(cMMI);
  257       }
  258       return nCStateRecs;
  259   
  260   } /* rtwCAPI_GetNumContStateRecords */

  288   /** Function: rtwCAPI_UpdateFullPaths =========================================*
  289    *
  290    */
  291   const char_T* rtwCAPI_UpdateFullPaths(rtwCAPI_ModelMappingInfo* mmi,
  292                                         const char_T* path,
  293                                         boolean_T isCalledFromTopModel)
  294   {
  295       int_T         i;
  296       int_T         nCMMI;
  297       size_t        pathLen;
  298       char_T*       mmiPath;
  299       size_t        mmiPathLen;
  300       char_T*       relMMIPath;
  301       size_t        relMMIPathLen;
  302   
  303       if (mmi == NULL) return NULL;
  304   
  305       utAssert(path != NULL);
  306       utAssert( rtwCAPI_GetFullPath(mmi) == NULL );
  307   
  308       pathLen = strlen(path)+1;
  309   
  310       if (isCalledFromTopModel) {
  311           /* If called from top model - FullPath is same as path */
  312           mmiPath = (char_T*)utMalloc(pathLen*sizeof(char_T));
  313           (void)memcpy(mmiPath, path, pathLen*sizeof(char_T));
  314       }
  315       else {        
  316           relMMIPath = rtwCAPI_EncodePath(rtwCAPI_GetPath(mmi));
  317           if ( (relMMIPath== NULL) && (rtwCAPI_GetPath(mmi) != NULL)) {
  318               return rtwCAPI_mallocError;
  319           }
  320           relMMIPathLen = relMMIPath ? (strlen(relMMIPath) + 1) : 0;
  321           
  322           mmiPathLen = pathLen + relMMIPathLen;
  323           
  324           mmiPath = (char_T*)utMalloc(mmiPathLen*sizeof(char_T));
  325           if (mmiPath == NULL) return rtwCAPI_mallocError;
  326           (void)memcpy(mmiPath, path, pathLen*sizeof(char_T));
  327           utAssert(mmiPath[pathLen-1] == '\0');
  328           
  329           if (relMMIPath) {
  330               /* mmiPath = path + | + relMMIPath + '\0' */
  331               mmiPath[pathLen-1] = '|';
  332               (void)memcpy(&(mmiPath[pathLen]),
  333                            relMMIPath, relMMIPathLen*sizeof(char_T));
  334               utAssert(mmiPath[mmiPathLen-1] == '\0');
  335               utFree(relMMIPath);
  336           }
  337       }
  338       rtwCAPI_SetFullPath(*mmi, mmiPath);
  339   
  340       nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);
  341       for (i = 0; i < nCMMI; ++i) {
  342           rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);
  343           const char_T* errstr = rtwCAPI_UpdateFullPaths(cMMI, mmiPath, 0);
  344           if (errstr != NULL) return errstr;
  345       }
  346       return NULL;
  347   
  348   } /* rtwCAPI_UpdateFullPaths */

  351   /** Function: rtwCAPI_GetFullStateBlockPath ===================================
  352    *
  353    */
  354   char* rtwCAPI_GetFullStateBlockPath(const char* stateBlockPath,
  355                                              const char* mmiPath,
  356                                              size_t      mmiPathLen,
  357                                              boolean_T   crossingModel)
  358   {
  359       char_T* blockPath          = NULL;
  360       char_T* fullStateBlockPath = NULL;
  361       size_t     fullStateBlockPathLen;
  362   
  363       if (stateBlockPath == NULL) goto EXIT_POINT;
  364   
  365       /* fullStateBlockPath = mmiPath + | + blockPath + '\0' */
  366       /* If crossing a model boundary encode, otherwise do not */
  367   
  368       if (crossingModel) {
  369           blockPath = rtwCAPI_EncodePath(stateBlockPath);
  370           if (blockPath == NULL) goto EXIT_POINT;
  371       } else {
  372           size_t len = strlen(stateBlockPath)+1;
  373           blockPath = (char*)utMalloc(len*sizeof(char));
  374           if (blockPath == NULL) goto EXIT_POINT;
  375           (void)strcpy(blockPath,stateBlockPath);
  376       }
  377       utAssert(blockPath != NULL);
  378       fullStateBlockPathLen = ( (mmiPath==NULL) ?
  379                                 strlen(blockPath) + 1 :
  380                                 mmiPathLen + strlen(blockPath) + 2 );
  381       fullStateBlockPath    = (char*)utMalloc(fullStateBlockPathLen*sizeof(char));
  382       if (fullStateBlockPath == NULL) goto EXIT_POINT;
  383   
  384       if (mmiPath != NULL) {
  385           (void)strcpy(fullStateBlockPath, mmiPath);
  386           fullStateBlockPath[mmiPathLen]   = '|';
  387           fullStateBlockPath[mmiPathLen+1] =  '\0';
  388           (void)strcat(fullStateBlockPath, blockPath);
  389       } else {
  390           (void)strcpy(fullStateBlockPath, blockPath);
  391           fullStateBlockPath[fullStateBlockPathLen-1] =  '\0';
  392       }
  393       utAssert(fullStateBlockPath[fullStateBlockPathLen-1] == '\0');
  394   
  395     EXIT_POINT:
  396       utFree(blockPath);
  397       return fullStateBlockPath; /* caller is responsible for free */
  398   }

  400   uint_T rtwCAPI_GetStateWidth(const rtwCAPI_DimensionMap* dimMap,
  401                                const uint_T*               dimArray,
  402                                const rtwCAPI_States*       states,
  403                                uint_T                      iState)
  404   {
  405       uint_T mapIdx = rtwCAPI_GetStateDimensionIdx(states, iState);
  406       utAssert( rtwCAPI_GetNumDims(dimMap,mapIdx) == 2 );
  407       mapIdx = rtwCAPI_GetDimArrayIndex(dimMap, mapIdx);
  408       return (dimArray[mapIdx] * dimArray[mapIdx+1]);
  409   }

  413   /** Function: rtwCAPI_GetStateRecordInfo =======================================
  414    *
  415    */
  416   const char_T* rtwCAPI_GetStateRecordInfo(const rtwCAPI_ModelMappingInfo* mmi,
  417                                            const char_T**    sigBlockName,
  418                                            const char_T**    sigLabel,
  419                                            const char_T**    sigName,
  420                                            int_T*            sigWidth,
  421                                            int_T*            sigDataType,
  422                                            int_T*            logDataType,
  423                                            int_T*            sigComplexity,
  424                                            void**            sigDataAddr,
  425                                            boolean_T*        sigCrossMdlRef,
  426                                            const char_T**    sigPathAlias,
  427                                            real_T*           sigSampleTime,
  428                                            int_T*            sigIdx,
  429                                            boolean_T         crossingModel,
  430                                            real_T*           contStateDeriv,
  431                                            boolean_T         rtwLogging)
  432   {
  433       int_T               i;
  434       int_T               nCMMI;
  435       int_T               nStates;
  436       const char_T*       mmiPath;
  437       size_t              mmiPathLen;
  438       const rtwCAPI_States*  states;
  439       const rtwCAPI_DimensionMap* dimMap;
  440       const uint_T*       dimArray;
  441       const rtwCAPI_DataTypeMap*  dataTypeMap;
  442       void**              dataAddrMap;
  443       const char_T*       errstr = NULL;
  444       uint8_T             isPointer = 0;
  445   
  446       if (mmi == NULL) goto EXIT_POINT;
  447   
  448       nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);
  449       for (i = 0; i < nCMMI; ++i) {
  450           rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);
  451           real_T* childContStateDeriv = NULL;
  452   
  453           if (cMMI == NULL) continue;
  454   
  455           if (contStateDeriv) {
  456               int idx;
  457   
  458               idx = rtwCAPI_MMIGetContStateStartIndex(cMMI);
  459               if(idx == -1) continue;
  460               
  461               childContStateDeriv = &contStateDeriv[idx];
  462           }
  463           errstr = rtwCAPI_GetStateRecordInfo(cMMI,
  464                                               sigBlockName,
  465                                               sigLabel,
  466                                               sigName,
  467                                               sigWidth,
  468                                               sigDataType,
  469                                               logDataType,
  470                                               sigComplexity,
  471                                               sigDataAddr,
  472                                               sigCrossMdlRef,
  473                                               sigPathAlias,
  474                                               sigSampleTime,
  475                                               sigIdx,
  476                                               0x1, /* true, */
  477                                               childContStateDeriv,
  478                                               rtwLogging);
  479           if (errstr != NULL) goto EXIT_POINT;
  480       }
  481   
  482       nStates = rtwCAPI_GetNumStates(mmi);
  483       if (nStates < 1) goto EXIT_POINT;
  484   
  485       mmiPath     = rtwCAPI_GetFullPath(mmi);
  486       mmiPathLen  = (mmiPath==NULL)? 0 : strlen(mmiPath);
  487       states      = rtwCAPI_GetStates(mmi);
  488       dimMap      = rtwCAPI_GetDimensionMap(mmi);
  489       dimArray    = rtwCAPI_GetDimensionArray(mmi);
  490       dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);
  491       dataAddrMap = rtwCAPI_GetDataAddressMap(mmi);
  492   
  493       for (i = 0; i < nStates; ++i) {
  494           uint_T      mapIdx;
  495   
  496           /* If collecting continuous states, skip non-continuous states */
  497           if (contStateDeriv && !rtwCAPI_IsAContinuousState(states,i)) continue;
  498   
  499           /* For RTW logging, skip states that cannot be logged to MAT-File. */
  500           if ((rtwLogging) &&
  501               (rtwCAPI_CanLogStateToMATFile(dataTypeMap, states, i) == false)) continue;
  502   
  503           /* BlockPath (caller is responsible for free) */
  504           sigBlockName[*sigIdx] =
  505               rtwCAPI_GetFullStateBlockPath(rtwCAPI_GetStateBlockPath(states,i),
  506                                             mmiPath, mmiPathLen, crossingModel);
  507           if (sigBlockName[*sigIdx] == NULL) {
  508               errstr = rtwCAPI_mallocError;
  509               goto EXIT_POINT;
  510           }
  511   
  512           /* Label */
  513           if (rtwCAPI_IsAContinuousState(states,i)){
  514               sigLabel[*sigIdx] = "CSTATE";
  515           } else {
  516               sigLabel[*sigIdx] = "DSTATE";
  517           }
  518           sigName[*sigIdx] = rtwCAPI_GetStateName(states, i);
  519   
  520           /* Width */
  521           sigWidth[*sigIdx] = rtwCAPI_GetStateWidth(dimMap, dimArray, states, i);
  522   
  523           /* DataType and logDataType */
  524           mapIdx = rtwCAPI_GetStateDataTypeIdx(states, i);
  525           sigDataType[*sigIdx] = rtwCAPI_GetDataTypeSLId(dataTypeMap, mapIdx);
  526           /* this mimics code in simulink.dll:DtGetDataTypeLoggingId() */
  527           if (logDataType) {
  528               switch (sigDataType[*sigIdx]) {
  529                 case SS_DOUBLE:
  530                 case SS_SINGLE:
  531                 case SS_INT8:
  532                 case SS_UINT8:
  533                 case SS_INT16:
  534                 case SS_UINT16:
  535                 case SS_INT32:
  536                 case SS_UINT32:
  537                 case SS_BOOLEAN:
  538                   logDataType[*sigIdx] = sigDataType[*sigIdx];
  539                   break;
  540                 case SS_ENUM_TYPE:
  541                   logDataType[*sigIdx] = SS_INT32;
  542                   break;
  543                 default:
  544                   logDataType[*sigIdx] = SS_DOUBLE;
  545                   break;
  546               }
  547           }
  548   
  549           /* Complexity */
  550           sigComplexity[*sigIdx] = rtwCAPI_GetDataIsComplex(dataTypeMap, mapIdx);
  551   
  552           /* Data Access - Pointer or Direct*/
  553           isPointer = ((uint8_T)rtwCAPI_GetDataIsPointer(dataTypeMap, mapIdx));
  554   
  555           /* Address */
  556           if (contStateDeriv) {
  557               int_T csvIdx = rtwCAPI_GetContStateStartIndex(states,i);
  558               utAssert(csvIdx >= 0);
  559               sigDataAddr[*sigIdx] = &contStateDeriv[csvIdx];
  560           } else {
  561               mapIdx = rtwCAPI_GetStateAddrIdx(states,i);
  562               rtwCAPI_GetSigAddrFromMap(isPointer, sigComplexity, sigDataType,
  563                                         sigDataAddr, sigIdx, mapIdx, dataAddrMap);
  564           }
  565   
  566           /* CrossingModelBoundary */
  567           sigCrossMdlRef[*sigIdx] = crossingModel;
  568   
  569           if (sigPathAlias && 
  570               rtwCAPI_GetStatePathAlias(states,i) != NULL && 
  571               rtwCAPI_GetStatePathAlias(states,i)[0] != '\0') {
  572               sigPathAlias[*sigIdx] =  
  573                   rtwCAPI_GetFullStateBlockPath(rtwCAPI_GetStatePathAlias(states,i),
  574                                                 mmiPath, mmiPathLen, crossingModel);
  575           }
  576           
  577           /* Sample Time */
  578           if (sigSampleTime) {
  579               const rtwCAPI_SampleTimeMap* tsMap = rtwCAPI_GetSampleTimeMap(mmi);
  580               int_T TID;
  581               mapIdx = rtwCAPI_GetStateSampleTimeIdx(states, i);
  582               TID = rtwCAPI_GetSampleTimeTID(tsMap, mapIdx);
  583               if (TID >= 0) {
  584                   sigSampleTime[2*(*sigIdx)] =
  585                       *((const real_T*)rtwCAPI_GetSamplePeriodPtr(tsMap,mapIdx));
  586                   sigSampleTime[2*(*sigIdx)+1] =
  587                       *((const real_T*)rtwCAPI_GetSampleOffsetPtr(tsMap,mapIdx));
  588               } else { /* triggered */
  589                   utAssert(TID==-1); 
  590                   sigSampleTime[2*(*sigIdx)]   = -1.0;
  591                   sigSampleTime[2*(*sigIdx)+1] = -1.0;
  592               }
  593           }
  594   
  595           ++(*sigIdx);
  596       }
  597   
  598     EXIT_POINT:
  599       return(errstr);
  600   
  601   } /* rtwCAPI_GetStateRecordInfo */

  663   /** Function: rtwCAPI_GetSigLogRecordInfo ======================================
  664    *
  665    */
  666   const char_T* rtwCAPI_GetSigLogRecordInfo(const rtwCAPI_ModelMappingInfo* mmi,
  667                                             const char_T**    sigBlockName,
  668                                             const char_T**    sigLabel,
  669                                             int_T*            sigWidth,
  670                                             int_T*            sigDataType,
  671                                             int_T*            logDataType,
  672                                             int_T*            sigComplexity,
  673                                             void**            sigDataAddr,
  674                                             boolean_T*        sigCrossMdlRef,
  675                                             int_T*            sigIdx,
  676                                             boolean_T         crossingModel,
  677                                             boolean_T         rtwLogging)
  678   {
  679       int_T               i;
  680       int_T               nCMMI;
  681       int_T               nSignals;
  682       const char_T*       mmiPath;
  683       size_t              mmiPathLen;
  684       const rtwCAPI_Signals*  signals;
  685       const rtwCAPI_DimensionMap* dimMap;
  686       const uint_T*       dimArray;
  687       const rtwCAPI_DataTypeMap*  dataTypeMap;
  688       void**              dataAddrMap;
  689       const char_T*       errstr = NULL;
  690       uint8_T             isPointer = 0;
  691       char*               blockPath = NULL;
  692   
  693       if (mmi == NULL) goto EXIT_POINT;
  694   
  695       nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);
  696       for (i = 0; i < nCMMI; ++i) {
  697           rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);
  698   
  699           errstr = rtwCAPI_GetSigLogRecordInfo(cMMI,
  700                                                sigBlockName,
  701                                                sigLabel,
  702                                                sigWidth,
  703                                                sigDataType,
  704                                                logDataType,
  705                                                sigComplexity,
  706                                                sigDataAddr,
  707                                                sigCrossMdlRef,
  708                                                sigIdx,
  709                                                true,
  710                                                rtwLogging);
  711           if (errstr != NULL) goto EXIT_POINT;
  712       }
  713   
  714       nSignals = rtwCAPI_GetNumSignals(mmi);
  715       if (nSignals < 1) goto EXIT_POINT;
  716   
  717       mmiPath     = rtwCAPI_GetFullPath(mmi);
  718       mmiPathLen  = (mmiPath==NULL)? 0 : strlen(mmiPath);
  719       signals     = rtwCAPI_GetSignals(mmi);
  720       dimMap      = rtwCAPI_GetDimensionMap(mmi);
  721       dimArray    = rtwCAPI_GetDimensionArray(mmi);
  722       dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);
  723       dataAddrMap = rtwCAPI_GetDataAddressMap(mmi);
  724   
  725       for (i = 0; i < nSignals; ++i) {
  726           uint_T mapIdx;
  727           size_t sigPathLen;
  728           char*  sigPath;
  729   
  730           /* For RTW logging, skip states that cannot be logged to MAT-File. */
  731           if ((rtwLogging) &&
  732               (rtwCAPI_CanLogSignalToMATFile(dataTypeMap, signals, i) == false)) continue;
  733   
  734           /* sigBlockPath = mmiPath + | + BlockPath + '\0' */
  735           /* If crossing a model boundary encode, otherwise do not */
  736   
  737           if (crossingModel) {
  738               blockPath = rtwCAPI_EncodePath(rtwCAPI_GetSignalBlockPath(signals, i));
  739               if ( (blockPath == NULL) &&
  740                    (rtwCAPI_GetSignalBlockPath(signals, i) != NULL)) {
  741                   errstr = rtwCAPI_mallocError;
  742                   goto EXIT_POINT;
  743               }
  744           } else {
  745               const char* constBlockPath = rtwCAPI_GetSignalBlockPath(signals, i);
  746               blockPath = (char*)utMalloc((strlen(constBlockPath)+1)*sizeof(char));
  747               (void)strcpy(blockPath, constBlockPath);
  748           }
  749           utAssert(blockPath != NULL);
  750           sigPathLen = ( (mmiPath==NULL) ?
  751                                      strlen(blockPath) + 1 :
  752                                      mmiPathLen + strlen(blockPath) + 2 );
  753           sigPath    = (char*)utMalloc(sigPathLen*sizeof(char));
  754           if (sigPath == NULL) {
  755               errstr = rtwCAPI_mallocError;
  756               goto EXIT_POINT;
  757           }
  758           if (mmiPath != NULL) {
  759               (void)strcpy(sigPath, mmiPath);
  760               sigPath[mmiPathLen]   = '|';
  761               sigPath[mmiPathLen+1] =  '\0';
  762               (void)strcat(sigPath, blockPath);
  763           } else {
  764               (void)strcpy(sigPath, blockPath);
  765               sigPath[sigPathLen-1] =  '\0';
  766           }
  767          /* need to free for every iteration of the loop, but also have
  768           * the free below EXIT_POINT in case of error */
  769           utFree(blockPath);
  770           blockPath = NULL;
  771           utAssert(sigPath[sigPathLen-1] == '\0');
  772           sigBlockName[*sigIdx] = sigPath; /* caller is responsible for free */
  773   
  774           /* Label */
  775           sigLabel[*sigIdx] = rtwCAPI_GetSignalName(signals, i);
  776   
  777           /* Width */
  778           mapIdx = rtwCAPI_GetSignalDimensionIdx(signals, i);
  779           utAssert( rtwCAPI_GetNumDims(dimMap,mapIdx) == 2 );
  780           mapIdx = rtwCAPI_GetDimArrayIndex(dimMap, mapIdx);
  781           sigWidth[*sigIdx] = dimArray[mapIdx] * dimArray[mapIdx+1];
  782   
  783           /* DataType and logDataType */
  784           mapIdx = rtwCAPI_GetSignalDataTypeIdx(signals, i);
  785           sigDataType[*sigIdx] = rtwCAPI_GetDataTypeSLId(dataTypeMap, mapIdx);
  786           /* this mimics code in simulink.dll:mapSigDataTypeToLogDataType */
  787           switch (sigDataType[*sigIdx]) {
  788             case SS_DOUBLE:
  789             case SS_SINGLE:
  790             case SS_INT8:
  791             case SS_UINT8:
  792             case SS_INT16:
  793             case SS_UINT16:
  794             case SS_INT32:
  795             case SS_UINT32:
  796             case SS_BOOLEAN:
  797               logDataType[*sigIdx] = sigDataType[*sigIdx];
  798               break;
  799             case SS_ENUM_TYPE:
  800               logDataType[*sigIdx] = SS_INT32;
  801               break;
  802             default:
  803               logDataType[*sigIdx] = SS_DOUBLE;
  804               break;
  805           }
  806   
  807           /* Complexity */
  808           sigComplexity[*sigIdx] = rtwCAPI_GetDataIsComplex(dataTypeMap, mapIdx);
  809   
  810           /* Data Access - Pointer or Direct*/
  811           isPointer = ((uint8_T)rtwCAPI_GetDataIsPointer(dataTypeMap, mapIdx));
  812   
  813           /* Address */
  814           mapIdx = rtwCAPI_GetSignalAddrIdx(signals, i);
  815   
  816           rtwCAPI_GetSigAddrFromMap(isPointer, sigComplexity, sigDataType,
  817                                     sigDataAddr, sigIdx, mapIdx, dataAddrMap);
  818   
  819           /* CrossingModelBoundary */
  820           sigCrossMdlRef[*sigIdx] = crossingModel;
  821   
  822           ++(*sigIdx);
  823       }
  824   
  825     EXIT_POINT:
  826       utFree(blockPath);
  827       return(errstr);
  828   
  829   } /* rtwCAPI_GetSigLogRecordInfo */

  832   /** Function: rtwCAPI_CountSysRan ==============================================
  833    *   Recursive function that counts the number of non-NULL pointers in the array
  834    *   of system ran dwork pointers, for the given MMI and below
  835    */
  836   void rtwCAPI_CountSysRan(const rtwCAPI_ModelMappingInfo *mmi,
  837   			 int                            *count)
  838   {
  839       sysRanDType **sysRan;
  840       int numSys;
  841       int nCMMI;
  842       int numSysRan = 0;
  843       int i;
  844   
  845       if (mmi == NULL) return;
  846   
  847       sysRan = rtwCAPI_GetSystemRan(mmi);
  848       numSys    = rtwCAPI_GetNumSystems(mmi);
  849       nCMMI     = rtwCAPI_GetChildMMIArrayLen(mmi);
  850   
  851       /* Recurse over children */
  852       for (i = 0; i < nCMMI; i++) {
  853           rtwCAPI_CountSysRan(rtwCAPI_GetChildMMI(mmi,i), &numSysRan);
  854       }
  855   
  856       /* Count number of dworks in this MMI - skip root */
  857       for (i = 1; i< numSys; i++) {
  858           if (sysRan[i] != NULL) numSysRan++;
  859       }
  860   
  861       *count += numSysRan;
  862   
  863   } /* end rtwCAPI_CountSysRan */

  866   /** Function: rtwCAPI_FillSysRan ===============================================
  867    *   Recursive function that fills in the system ran dwork pointers and their
  868    *   corresponding tids in the given array, for the given MMI and below.  The
  869    *   array to be filled in must be allocated outside
  870    */
  871   void rtwCAPI_FillSysRan(const rtwCAPI_ModelMappingInfo *mmi,
  872   			sysRanDType                    **sysRan,
  873   			int                            *sysTid,
  874                           int                            *fillIdx)
  875   {
  876       int       numSys;
  877       sysRanDType **mmiSysRan;
  878       int       nCMMI;
  879       int       *mmiSysTid;
  880       int       idx         = *fillIdx;
  881       const int *mmiConSys;
  882       int i;
  883   
  884       if (mmi == NULL) return;
  885   
  886       numSys      = rtwCAPI_GetNumSystems(mmi);
  887       mmiSysRan = rtwCAPI_GetSystemRan(mmi);
  888       nCMMI       = rtwCAPI_GetChildMMIArrayLen(mmi);
  889       mmiSysTid  = rtwCAPI_GetSystemTid(mmi);
  890       mmiConSys  = rtwCAPI_GetContextSystems(mmi);
  891   
  892       /* Recurse over children */
  893       for (i = 0; i < nCMMI; i++) {
  894           rtwCAPI_FillSysRan(rtwCAPI_GetChildMMI(mmi,i), sysRan, sysTid, &idx);
  895       }
  896   
  897       /* Populate arrays with dwork pointers and tid - skip root */
  898       for (i = 1; i< numSys; i++) {
  899           if (mmiSysRan[i] != NULL) {
  900               idx++;
  901               sysRan[idx] = mmiSysRan[i];
  902               sysTid[idx] = mmiSysTid[mmiConSys[i]];
  903           }
  904       }
  905   
  906       *fillIdx = idx;
  907   
  908   } /* end rtwCAPI_FillSysRan */
